```python
def fibonacci(n):
    """Calculate Fibonacci"""
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# Print result
for i in range(5):
    print(f"F({i}) = {fibonacci(i)}")
```
